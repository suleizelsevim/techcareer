## npm sıklıkla kullanılan komutlar
```sh

npm search express
npm install express
npm install express@1.15.3
npm install express -g #indirilen paket global ortamda çalışır yoksa localde çalışır
npm install express -D
npm install express # local
npm install express --save # local
npm install express --save-dev # -dev sadece geliştirme ortamında çalışır

npm uninstall express
npm uninstall express@1.15.3

npm list
npm list -g # globalde hangi paketleri yüklemişim
npm list -g --depth=0 # globalde sadece ana branche yüklediklerimi göster

npm root # local projedeki node_modules kütüphanesini gösterir
npm root -g # global projedeki node_modules kütüphanesini gösterir

npm install
npm i
npm update
npm update express

npm start
npm test
```
